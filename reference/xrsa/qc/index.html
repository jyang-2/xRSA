    def compute_trial_rdm_blockavg(ds_trial_rdm, mask=None):
    """Computes block averages of trial RDMs, using 'trial_row' and 'trial_col'.

    'trial_row' and 'trial_col' should be multiindices, with coords (row_stim, row_stim_occ).

    Args:
        ds_trial_rdm (xr.Dataset): should have dims `trial_row` and `trial_col` as multiindices,
          with coords `row(col)_stim` and `row(col)_stim_occ`. Additional dimensions are okay.
        mask (xr.DataArray): mask for everything except the block identity lower triangular
          elements. If one is not provided, it will be automatically computed using
          `identity_block_tril`.

    Returns:
        ds_trial_rdm_blockavg (xr.Dataset):
    """
    if mask is None:
        mask = ~identity_block_tril(ds_trial_rdm)

    masked_trial_rdm = ds_trial_rdm.where(mask)
    masked_trial_rdm = (masked_trial_rdm
                        .reset_index(['trial_row', 'trial_col'])
                        .set_index(trial_row='row_stim')
                        .set_index(trial_col='col_stim')
                        )
    ds_trial_rdm_blockavg = xarray_reduce(masked_trial_rdm,
                                          'trial_row', 'trial_col',
                                          func='mean')
    return ds_trial_rdm_blockavg

    def identity_block_tril(ds_trial_rdm):
    """Get a mask for the lower triangular elements of the block diagonals of a trial RDM"""
    mask_block_diag = ds_trial_rdm['row_stim'] == ds_trial_rdm['col_stim']
    mask_block_tril = ds_trial_rdm['row_stim_occ'] >= ds_trial_rdm['col_stim_occ']
    mask = mask_block_tril & mask_block_diag

    return mask

    def qcthresh_by_blockavg(trial_rdm_concat,
                         stim_rdm_concat,
                         trial_rdm_concat_blockavg,
                         sim_threshold,
                         ):
    """Loop through acquisitiions, and check the diagonal values."""

    trial_rdm_concat_blockavg0 = trial_rdm_concat_blockavg.copy(deep=True)
    trial_rdm_concat0 = trial_rdm_concat.copy(deep=True)
    stim_rdm_concat0 = stim_rdm_concat.copy(deep=True)

    n_acq = trial_rdm_concat_blockavg0.dims['acq']
    stim_list = trial_rdm_concat_blockavg0['trial_row'].to_numpy()

    for iacq in range(n_acq):
        for stim in stim_list:
            block_avg_stim = 1 - trial_rdm_concat_blockavg0.sel(acq=iacq,
                                                                trial_row=stim,
                                                                trial_col=stim).item()
            if ~np.isnan(block_avg_stim) and block_avg_stim < sim_threshold:
                print('dropping: '
                      f'\n\t- iacq = {iacq}'
                      f'\n\t- stim = {stim}'
                      f'\n\t- block_avg = {block_avg_stim}')

                # nan out the rows and columns of stim
                trial_rdm_concat_blockavg0.loc[dict(acq=iacq, trial_row=stim)] = np.nan
                trial_rdm_concat_blockavg0.loc[dict(acq=iacq, trial_col=stim)] = np.nan

                trial_rdm_concat0.loc[dict(acq=iacq, row_stim=stim)] = np.nan
                trial_rdm_concat0.loc[dict(acq=iacq, col_stim=stim)] = np.nan

                stim_rdm_concat0.loc[dict(acq=iacq, stim_row=stim)] = np.nan
                stim_rdm_concat0.loc[dict(acq=iacq, stim_col=stim)] = np.nan

    trial_rdm_concat_blockavg0.attrs['qc.sim_threshold'] = sim_threshold
    trial_rdm_concat0.attrs['qc.sim_threshold'] = sim_threshold
    stim_rdm_concat0.attrs['qc.sim_threshold'] = sim_threshold

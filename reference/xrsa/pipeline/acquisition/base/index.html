class AcquisitionAnalysis:
    acq: Acquisition = field(converter=Acquisition.from_stat_file)
    params: PipelineParams = field(converter=PipelineParams.from_dict)
    ds_suite2p_outputs: xr.Dataset = field(init=False, default=None)
    ds_trials: xr.Dataset = field(init=False, default=None)
    ds_bc_trials: xr.Dataset = field(init=False, default=None)
    ds_peak_amp: xr.Dataset = field(init=False, default=None)
    ds_baseline_std: xr.Dataset = field(init=False, default=None)
    ds_peak_amp_stim: xr.Dataset = field(init=False, default=None)
    ds_bc_trials_stimavg: xr.Dataset = field(init=False, default=None)
    ds_trials_stimavg: xr.Dataset = field(init=False, default=None)
    ds_stimavg_peak_amp: xr.Dataset = field(init=False, default=None)
    ds_stimavg_baseline_std: xr.Dataset = field(init=False, default=None)
    multi_rdm_calculator: MultiRdmCalculator = field(init=False, default=None)
    ds_trial_rdm: xr.Dataset = field(init=False)
    ds_stim_rdm: xr.Dataset = field(init=False)
    ds_stimavg_rdm: xr.Dataset = field(init=False)
    ds_blockavg_rdm: xr.Dataset = field(init=False)

    def make_suite2p_base(self):
        if len(self.acq.timestamps) == 0:
            self.acq.load_timestamps()
        if len(self.acq.stim_list) == 0:
            self.acq.load_stim_list()

        ds_suite2p_outputs = external.suite2p.convert.outputs_2_xarray_base(self.acq.stat_file)

        ds_suite2p_outputs = xrsa.timeseries.add_timestamps_to_suite2p_outputs(
                ds_suite2p_outputs,
                timestamps=self.acq.timestamps['stack_times']
                )
        # add attrs
        dacq = self.acq.to_dict()
        ds_suite2p_outputs.attrs = {f"acq.{k}": v for k, v in dacq.items()}
        self.ds_suite2p_outputs = ds_suite2p_outputs

    def add_iscell_files(self):
        iscell_params = self.params.iscell_params
        if len(iscell_params.iscell_filenames) > 0:
            iscell_coords = {}
            for filename in iscell_params.iscell_filenames:
                iscell_filepath = self.acq.stat_file.with_name(filename)
                iscell = np.load(iscell_filepath)
                iscell_coords[iscell_filepath.stem] = ('cells', iscell[:, 0].astype(int))

            self.ds_suite2p_outputs = self.ds_suite2p_outputs.assign_coords(
                    iscell_coords
                    )
            self.ds_suite2p_outputs.attrs['iscell.iscell_filenames'] = (
                iscell_params.iscell_filenames)

    def add_rmap_files(self):
        rmap_params = self.params.rmap_params

        if len(rmap_params.rmap_filenames) > 0:
            rmap_coords = {}
            for filename in rmap_params.rmap_filenames:
                rmap_filepath = self.acq.stat_file.with_name(filename)
                stem = rmap_filepath.stem

                rmap = np.load(rmap_filepath)
                rmap_coords[f"{stem}_cluster"] = ('cells', rmap[:, 0])
                rmap_coords[f"{stem}_emb"] = ('cells', rmap[:, 1])

            self.ds_suite2p_outputs = self.ds_suite2p_outputs.assign_coords(
                    rmap_coords
                    )
            self.ds_suite2p_outputs.attrs['rmap.rmap_filenames'] = rmap_params.rmap_filenames

    def make_trials(self):
        start, stop, step = self.params.trial_params.trial_ts
        trial_ts = np.arange(start, stop + 1e-6, step)

        self.ds_trials = xrsa.trials.timeseries_2_trials(
                self.ds_suite2p_outputs,
                stim_ict=self.acq.timestamps['olf_ict'],
                stim_list=self.acq.stim_list,
                trial_ts=trial_ts,
                index_stimuli=True,
                stimulus_index_keys=self.params.trial_params.stimulus_index_keys)

    def make_baseline_trials(self):
        trial_baseline_params = self.params.trial_baseline_params

        self.ds_bc_trials = xrsa.trials.baseline_correct_trials(
                self.ds_trials,
                baseline_win=trial_baseline_params.baseline_win,
                baseline_method=trial_baseline_params.baseline_method,
                baseline_quantile=trial_baseline_params.baseline_quantile
                )

    def average_trials_by_stim(self):
        trial_baseline_params = self.params.trial_baseline_params

        if self.ds_trials is not None:
            self.ds_trials_stimavg = self.ds_trials.groupby('stim').mean(dim='trials')

        if self.ds_bc_trials is not None:
            ds_bc_trials_stimavg = self.ds_bc_trials.groupby('stim').mean(dim='trials')
            ds_bc_trials_stimavg = xrsa.trials.baseline_correct_trials(
                    ds_bc_trials_stimavg,
                    baseline_win=trial_baseline_params.baseline_win,
                    baseline_method=trial_baseline_params.baseline_method,
                    baseline_quantile=trial_baseline_params.baseline_quantile
                    )
            self.ds_bc_trials_stimavg = ds_bc_trials_stimavg

    def make_peak_amp(self):
        peak_amp_params = self.params.peak_amp_params
        self.ds_peak_amp = xrsa.respvec.peak_amp(
                self.ds_bc_trials,
                peak_win=peak_amp_params.peak_win,
                peak_method=peak_amp_params.peak_method,
                subtract_baseline=peak_amp_params.subtract_baseline,
                baseline_win=peak_amp_params.baseline_win,
                baseline_method=peak_amp_params.baseline_method)

        self.ds_stimavg_peak_amp = xrsa.respvec.peak_amp(
                self.ds_bc_trials_stimavg,
                peak_win=peak_amp_params.peak_win,
                peak_method=peak_amp_params.peak_method,
                subtract_baseline=peak_amp_params.subtract_baseline,
                baseline_win=peak_amp_params.baseline_win,
                baseline_method=peak_amp_params.baseline_method)

    def make_peak_amp_stim(self):
        self.ds_peak_amp_stim = self.ds_peak_amp.groupby('stim').mean(dim='trials')

    def make_baseline_std(self):
        baseline_std_dev_params = self.params.baseline_std_dev_params
        self.ds_baseline_std = xrsa.respvec.baseline_std(
                self.ds_bc_trials,
                baseline_std_win=baseline_std_dev_params.baseline_std_win
                )

        self.ds_stimavg_baseline_std = xrsa.respvec.baseline_std(
                self.ds_bc_trials_stimavg,
                baseline_std_win=baseline_std_dev_params.baseline_std_win
                )

    def masked_trial_rdm(self, cell_mask_coord=None, metric='correlation',
                         # assign_coords=False
                         ):
        if cell_mask_coord is not None:
            cell_mask = self.ds_peak_amp[cell_mask_coord]
            ds_peak_amp = self.ds_peak_amp.where(cell_mask > 0, drop=True)

        else:
            ds_peak_amp = self.ds_peak_amp

        n_rois = ds_peak_amp.sizes['cells']

        ds_masked_trial_rdm = xrsa.rdm.compute_trial_respvec_rdm(
                ds_peak_amp,
                metric=metric).assign_coords(cell_mask_coord=cell_mask_coord,
                                             n_rois=n_rois,
                                             metric=metric
                                             )

        return ds_masked_trial_rdm

    def masked_stim_rdm(self, cell_mask_coord=None, metric='correlation'):
        if cell_mask_coord is not None:
            cell_mask = self.ds_peak_amp_stim[cell_mask_coord]
            ds_peak_amp_stim = self.ds_peak_amp_stim.where(cell_mask > 0, drop=True)

        else:
            ds_peak_amp_stim = self.ds_peak_amp_stim

        n_rois = ds_peak_amp_stim.sizes['cells']

        ds_masked_stim_rdm = xrsa.rdm.compute_rdm(
                ds_peak_amp_stim,
                metric=metric,
                input_dim_ord=['stim', 'cells']
                ).assign_coords(
                cell_mask_coord=cell_mask_coord,
                n_rois=n_rois,
                metric=metric
                )

        return ds_masked_stim_rdm

    def masked_stim_rdm_from_stimavg(self, cell_mask_coord=None, metric='correlation'):
        if cell_mask_coord is not None:
            cell_mask = self.ds_stimavg_peak_amp[cell_mask_coord]
            ds_peak_amp_stim = self.ds_stimavg_peak_amp.where(cell_mask > 0, drop=True)

        else:
            ds_peak_amp_stim = self.ds_stimavg_peak_amp

        n_rois = ds_peak_amp_stim.sizes['cells']

        ds_masked_stim_rdm = xrsa.rdm.compute_rdm(
                ds_peak_amp_stim,
                metric=metric,
                input_dim_ord=['stim', 'cells']
                ).assign_coords(
                cell_mask_coord=cell_mask_coord,
                n_rois=n_rois,
                metric=metric
                )

        return ds_masked_stim_rdm

    def make_multi_rdm_calulator(self):
        multi_rdm_calculator_params = self.params.multi_rdm_calculator_params

        self.multi_rdm_calculator = MultiRdmCalculator(
                ds_peak_amp=self.ds_peak_amp,
                ds_peak_amp_stim=self.ds_peak_amp_stim,
                ds_stimavg_peak_amp=self.ds_stimavg_peak_amp,
                cell_mask_coords=multi_rdm_calculator_params.cell_mask_coords,
                metrics=multi_rdm_calculator_params.metrics
                )

    def compute_multi_rdms(self, sort_stim=True):
        ds_trial_rdm = (self.multi_rdm_calculator.multicompute_masked_rdm(which='trial')
                        # .sortby('trial_row')
                        # .sortby('trial_col')
                        )
        ds_stim_rdm = (self.multi_rdm_calculator.multicompute_masked_rdm(which='stim')
                       # .sortby('stim_row')
                       # .sortby('stim_col')
                       )
        ds_stimavg_rdm = (self.multi_rdm_calculator.multicompute_masked_rdm(which='stimavg')
                          # .sortby('stim_row')
                          # .sortby('stim_col')
                          )
        if sort_stim:
            ds_trial_rdm = ds_trial_rdm.sortby('trial_row').sortby('trial_col')
            ds_stim_rdm = ds_stim_rdm.sortby('stim_row').sortby('stim_col')
            ds_stimavg_rdm = ds_stimavg_rdm.sortby('stim_row').sortby('stim_col')

        self.ds_trial_rdm = ds_trial_rdm
        self.ds_stim_rdm = ds_stim_rdm
        self.ds_stimavg_rdm = ds_stimavg_rdm
        self.ds_blockavg_rdm = xrsa.qc.compute_trial_rdm_blockavg(ds_trial_rdm)

    def save_trials(self, acq_dir):
        """Saves trial (cell x time x trial or stim) to netcdfs in {{acq_dir}}/trials

        Trial structures saved are:
          - ds_trials
          - ds_bc_trials
          - ds_bc_trials_stimavg
          - ds_trials_stimavg
        """
        acq_dir.joinpath('trials').mkdir(exist_ok=True, parents=True)

        self.ds_trials.reset_index('trials').to_netcdf(
                acq_dir.joinpath('trials', 'xrds_trials.nc'),
                )
        self.ds_bc_trials.reset_index('trials').to_netcdf(
                acq_dir.joinpath('trials', 'xrds_bc_trials.nc'),
                )
        self.ds_trials_stimavg.to_netcdf(acq_dir.joinpath('trials', 'xrds_trials_stimavg.nc'))
        self.ds_bc_trials_stimavg.to_netcdf(acq_dir.joinpath('trials', 'xrds_bc_trials_stimavg.nc'))

    def save_respvecs(self, acq_dir):
        """Saves peak amplitudes to netcdfs in {{acq_dir}}/respvec

        Respvec datasets saved are:
            - ds_peak_amp: xr.Dataset = field(init=False, default=None)
            - ds_baseline_std: xr.Dataset = field(init=False, default=None)
            - ds_peak_amp_stim: xr.Dataset = field(init=False, default=None)
            - ds_stimavg_peak_amp: xr.Dataset = field(init=False, default=None)
            - ds_stimavg_baseline_std: xr.Dataset = field(init=False, default=None)
        """
        acq_dir.joinpath('respvec').mkdir(exist_ok=True, parents=True)

        self.ds_peak_amp.reset_index('trials').to_netcdf(
                acq_dir.joinpath('respvec', 'xrds_peak_amp.nc')
                )
        self.ds_baseline_std.reset_index('trials').to_netcdf(
                acq_dir.joinpath('respvec', 'xrds_baseline_std.nc')
                )
        self.ds_peak_amp_stim.to_netcdf(acq_dir.joinpath('respvec', 'xrds_peak_amp_stim.nc'))

        self.ds_stimavg_peak_amp.to_netcdf(
                acq_dir.joinpath('respvec', 'xrds_stimavg_peak_amp.nc')
                )
        self.ds_stimavg_baseline_std.to_netcdf(
                acq_dir.joinpath('respvec', 'xrds_stimavg_baseline_std.nc')
                )

    def save_rdms(self, acq_dir):
        acq_dir.joinpath('trial_rdm').mkdir(parents=True, exist_ok=True)

        # save trial rdms
        (self.ds_trial_rdm
         .reset_index('trial_col')
         .reset_index('trial_row')
         .reset_index('rdms')
         .to_netcdf(acq_dir.joinpath('trial_rdm', 'xrds_trial_rdm.nc'))
         )
        (self.ds_blockavg_rdm
         .reset_index('rdms')
         .to_netcdf(acq_dir.joinpath('trial_rdm', 'xrds_blockavg_rdm.nc'))
         )

        # save stim rdms
        acq_dir.joinpath('stim_rdm').mkdir(parents=True, exist_ok=True)
        (self.ds_stim_rdm
         .reset_index('rdms')
         .to_netcdf(acq_dir.joinpath('stim_rdm', 'xrds_stim_rdm.nc'))
         )
        (self.ds_stimavg_rdm
         .reset_index('rdms')
         .to_netcdf(acq_dir.joinpath('stim_rdm', 'xrds_stimavg_rdm.nc'))
         )
        print('rdms saved.')

    def save(self, acq_dir):
        if acq_dir.parents[1].name != 'data_by_acq_2':
            raise ValueError("Wrong parent directory (2nd parent must be 'data_by_acq_2)")

        acq_dir.mkdir(parents=True, exist_ok=True)
        self.save_trials(acq_dir)
        self.save_respvecs(acq_dir)
        self.save_rdms(acq_dir)
        self.params.to_yaml(acq_dir.joinpath('pipeline_params.yaml'))

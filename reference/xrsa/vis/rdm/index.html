    def fig_size_calculator(n_rows, n_cols,
                        ax_width=2.5,
                        ax_height=2.5,
                        axes_pad=0.25,
                        left_margin=1.25,
                        right_margin=1,
                        bottom_margin=1.25,
                        top_margin=0.5,
                        ):
    """

    Args:
        n_rows (int): number of rows
        n_cols (int): number of axis columns
        ax_width: axis width in inches
        ax_height: axis height in inches
        axes_pad: spacing between axes
        left_margin: left figure margin in inches
        right_margin: right figure margin in inches
        bottom_margin: bottom margin in inches
        top_margin: bottom margin in inches

    Returns:
        figsize (tuple): (width, height) of figure
        subplot_kws (dict): kwargs for plt.subplots_adjust (left, right, top, bottom)
    """

    axes_width = (ax_width + axes_pad) * n_cols
    fig_width = axes_width + left_margin + right_margin

    axes_height = (ax_height + axes_pad) * n_rows
    fig_height = axes_height + bottom_margin + top_margin

    fig_size = (fig_width, fig_height)
    subplot_adjust_kws = dict(left=left_margin / fig_width,
                              right=1 - right_margin / fig_width,
                              top=1 - top_margin / fig_height,
                              bottom=bottom_margin / fig_height)
    return fig_size, subplot_adjust_kws

    def plot_mean_and_std_rdms(da_rsm_mean,
                           da_rsm_std,
                           row_coord='row_stim',
                           col_coord='col_stim',
                           heatmap_kws=None,
                           fig_size_calculator_kws=None,
                           ):
    """
    Plot mean and std. trial RDM heatmaps.

    The mean and std of the trial RDM should be computed from a `trial_rsm_concat` xr.Dataset.
    See `aggregate_rdms.py`

    Args:
        da_rsm_mean (xr.DataArray): average trial RSM
        da_rsm_std: st. dev. RSM
        col_coord: if trial RDM, row_stim; if stim RDM, stim_row
        row_coord: if trial RDM, row_stim; if stim RDM, stim_row
        heatmap_kws:
        fig_size_calculator_kws (dict, optional):
            kws for computing figure size in `fig_size_calculator`.

    Returns:
        fig
        grid
    """
    if fig_size_calculator_kws is None:
        figsize, subplots_adjust_kws = fig_size_calculator(1, 2)
    else:
        figsize, subplots_adjust_kws = fig_size_calculator(1, 2, **fig_size_calculator_kws)

    fig = plt.figure(figsize=figsize)
    grid = ImageGrid(fig, 111,  # similar to subplot(111)
                     nrows_ncols=(1, 2),  # creates 2x2 grid of axes
                     axes_pad=0.25,  # pad between axes in inch.
                     cbar_mode='edge',
                     cbar_size='5%',
                     cbar_pad=0.25
                     )

    # plot mean RDM
    for ax, da in zip(grid.axes_all, (da_rsm_mean, da_rsm_std)):
        plot_rsm_heatmap(da_rsm_mean,
                         row_coord=row_coord,
                         col_coord=col_coord,
                         ax=ax,
                         cbar_ax=ax.cax,
                         heatmap_kws=heatmap_kws)
        ax.tick_params(axis='both', labelsize=8)
        ax.set_xlabel('')
        ax.set_ylabel('')

    grid.axes_all[0].set_title('mean', fontsize=8)
    grid.axes_all[1].set_title('std', fontsize=8, )
    plt.subplots_adjust(**subplots_adjust_kws)
    return fig, grid

    def plot_trial_and_stim_summary_rsms(da_trial_rsm_mean,
                                     da_trial_rsm_std,
                                     da_stim_rsm_mean,
                                     da_stim_rsm_std,
                                     da_blockavg_mean=None,
                                     da_blockavg_std=None,
                                     trial_coords=('row_stim', 'col_stim'),
                                     stim_coords=('stim_row', 'stim_col'),
                                     blockavg_coords=('trial_row', 'trial_col'),
                                     n_acqs=None,
                                     heatmap_kws=None,
                                     fig_size_calculator_kws=None,
                                     ):
    """Plot mean and std. trial RDM heatmaps.

    The mean and std of the trial RDM should be computed from a `trial_rsm_concat` xr.Dataset.
    See `aggregate_rdms.py`

    Args:

        da_trial_rsm_mean (xr.DataArray): average trial RSM
        da_trial_rsm_std (xr.DataArray): stdev trial RSM
        da_stim_rsm_std (xr.DataArray): mean stim RSM
        da_stim_rsm_mean (xr.DataArray): stdev stim RSM
        da_blockavg_mean (xr.DataArray, optional): mean blockavg. RDM
        da_blockavg_std (xr.DataArray, optional): stdev blockavg. RDM
        trial_coords: (row_coord, col_coord) names for trial RDMs
        stim_coords: (row_coord, col_coord) names for stim RDMs
        blockavg_coords: (row_coord, col_coord) names from blockavg. RDMs
        heatmap_kws (dict): keyword arguments for plot_sub:
        fig_size_calculator_kws (dict, optional):
            kws for computing figure size in `fig_size_calculator`.
        n_acqs: # of acquisitions (to annote axes)

    Returns:
        fig
        grid
    """
    include_blockavg = not ((da_blockavg_mean is None) & (da_blockavg_std is None))

    if include_blockavg:
        n_rows = 2
        n_cols = 3
    else:
        n_rows = 2
        n_cols = 2

    print(f"include_blockavg = {include_blockavg}"
          f"\nn_rows = {n_rows}"
          f"\nn_cols = {n_cols}"
          )

    # compute figure size params
    if fig_size_calculator_kws is None:
        figsize, subplots_adjust_kws = fig_size_calculator(n_rows, n_cols)
    else:
        figsize, subplots_adjust_kws = fig_size_calculator(n_rows, n_cols,
                                                           **fig_size_calculator_kws)

    print(f"figsize = {figsize}")
    print("subplots_adjust_kws:")
    print(subplots_adjust_kws)

    fig = plt.figure(figsize=figsize)

    # grid = ImageGrid(fig, 111,  # similar to subplot(111)
    #                  nrows_ncols=(n_rows, n_cols),  # creates 2x2 grid of axes
    #                  axes_pad=0.25,  # pad between axes in inch.
    #                  cbar_mode='edge',
    #                  cbar_size='5%',
    #                  cbar_pad=0.25,
    #                  share_all=False,
    #                  aspect=False
    #                  )

    grid = Grid(fig, 111,  # similar to subplot(111)
                nrows_ncols=(n_rows, n_cols),  # creates 2x2 grid of axes
                axes_pad=0.25,  # pad between axes in inch.
                share_all=False,
                share_x=False,
                share_y=False,
                )
    caxs = [item.inset_axes([1.1, 0, 0.05, 1]) for item in grid.axes_column[-1]]

    # plot trial RDMs
    for ax, cax, da in zip(grid.axes_column[0],
                           caxs,
                           (da_trial_rsm_mean, da_trial_rsm_std)):
        plot_rsm_heatmap(da,
                         row_coord=trial_coords[0],
                         col_coord=trial_coords[1],
                         ax=ax,
                         cbar_ax=cax,
                         heatmap_kws=heatmap_kws)
        ax.tick_params(axis='both', labelsize=8)
        ax.set_xlabel('')
        ax.set_ylabel('')

    grid.axes_row[0][0].set_title('trial', fontsize=10)

    # plot stim RDMs
    for ax, cax, da in zip(grid.axes_column[1],
                           caxs,
                           (da_stim_rsm_mean, da_stim_rsm_std)):
        plot_rsm_heatmap(da,
                         row_coord=stim_coords[0],
                         col_coord=stim_coords[1],
                         ax=ax,
                         cbar_ax=cax,
                         heatmap_kws=heatmap_kws)
        ax.tick_params(axis='both', labelsize=8)
        ax.set_xlabel('')
        ax.set_ylabel('')
    grid.axes_row[0][1].set_title('stim', fontsize=10)

    # plot blockavg. RDMs
    if include_blockavg:
        for ax, cax, da in zip(grid.axes_column[2],
                               caxs,
                               (da_blockavg_mean, da_blockavg_std)):
            plot_rsm_heatmap(da,
                             row_coord=blockavg_coords[0],
                             col_coord=blockavg_coords[1],
                             ax=ax,
                             cbar_ax=cax,
                             heatmap_kws=heatmap_kws)
            ax.tick_params(axis='both', labelsize=8)
            ax.set_xlabel('')
            ax.set_ylabel('')
        grid.axes_row[0][2].set_title('block avg.', fontsize=10)

    grid.axes_row[0][0].set_ylabel('mean', fontsize=10)
    grid.axes_row[1][0].set_ylabel('std', fontsize=10)

    for ax in grid.axes_all:
        ax.set(facecolor='0.7')

    # annotate w/ n_acqs
    if n_acqs is not None:
        annot_axes_with_n(grid.axes_row[0], [n_acqs] * n_cols)

    plt.subplots_adjust(**subplots_adjust_kws)

    return fig, grid
